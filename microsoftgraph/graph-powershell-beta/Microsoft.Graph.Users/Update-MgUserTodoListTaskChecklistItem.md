```
 -UserId <String> [-AdditionalProperties <Hashtable>] [-CheckedDateTime <DateTime>]
 [-CreatedDateTime <DateTime>] [-DisplayName <String>] [-Id <String>] [-IsChecked] [-PassThru] [-WhatIf]
 [-Confirm] [<CommonParameters>]
```
